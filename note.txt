Schema (Hybrid: Strong Columns + JSONB Objects)

CREATE TABLE job_execution_event (
    -- Identity
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

    -- Execution identity
    execution_id          UUID NOT NULL UNIQUE,
    parent_execution_id   UUID NULL,

    -- Job metadata
    job_name              VARCHAR(100) NOT NULL,
    job_type              VARCHAR(30)  NOT NULL,   -- BATCH | WEBHOOK
    trigger_type           VARCHAR(30)  NOT NULL,  -- SCHEDULED | MANUAL | API | FILE_DROP
    source_system         VARCHAR(50)  NOT NULL,

    -- Lifecycle
    status                VARCHAR(30)  NOT NULL,   -- RECEIVED | PROCESSING | PARTIAL | SUCCESS | FAILED
    stage                 VARCHAR(50)  NOT NULL,

    -- Aggregate counts (execution-level truth)
    total_files            INTEGER DEFAULT 0,
    total_records          INTEGER DEFAULT 0,
    processed_records      INTEGER DEFAULT 0,
    success_records        INTEGER DEFAULT 0,
    failed_records         INTEGER DEFAULT 0,

    -- File-level details (OBJECT)
    file_execution_details JSONB NOT NULL,

    -- Error summary (OBJECT)
    error_summary          JSONB NULL,

    -- Versioning
    job_version            VARCHAR(20) NULL,

    -- Traceability
    correlation_id         UUID NOT NULL,

    -- Timing
    started_at             TIMESTAMP WITH TIME ZONE NOT NULL,
    completed_at           TIMESTAMP WITH TIME ZONE NULL,
    duration_ms            BIGINT NULL,

    -- Audit
    created_by             VARCHAR(50) NOT NULL,
    created_at             TIMESTAMP WITH TIME ZONE DEFAULT now()
);
