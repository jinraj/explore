CREATE TABLE member_activity (
    -- Surrogate identity
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

    -- Natural business key (idempotency)
    member_id                VARCHAR(50) NOT NULL,
    icario_enrollment_id     UUID NOT NULL,
    program_id               UUID NOT NULL,
    activity_code            VARCHAR(50) NOT NULL,

    -- Activity linkage
    contact_event_id         UUID NULL,        -- links OAF ↔ ARF
    survey_id                UUID NULL,        -- if activity is survey-based

    -- Activity outcome
    activity_status          VARCHAR(30) NOT NULL,  -- COMPLETED | INCOMPLETE | INVALID
    activity_completed_at    TIMESTAMP WITH TIME ZONE NULL,

    -- Source & timing
    activity_source          VARCHAR(30) NOT NULL,  -- BATCH | API
    source_event_timestamp   TIMESTAMP WITH TIME ZONE NOT NULL,

    -- Enrollment context (important for rewards)
    enrollment_event_type    VARCHAR(30) NULL,      -- New | Renewal | Cancel
    enrollment_status        VARCHAR(30) NULL,      -- ENROLLED | DISENROLLED

    -- Segmentation / business context
    protocol                 VARCHAR(20) NULL,
    reporting_segment        VARCHAR(20) NULL,
    plan_contract_id         VARCHAR(50) NULL,

    -- Execution traceability
    execution_id             UUID NOT NULL,          -- from job_execution_event
    correlation_id           UUID NOT NULL,

    -- Audit
    created_at               TIMESTAMP WITH TIME ZONE DEFAULT now(),

    -- Idempotency protection
    CONSTRAINT uq_member_activity
        UNIQUE (
            member_id,
            icario_enrollment_id,
            program_id,
            activity_code
        )
);


CREATE TABLE reward_calculation (
    -- Surrogate identity
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

    -- Activity reference (the anchor)
    member_activity_id UUID NOT NULL,

    -- Natural idempotency (activity + rule version)
    reward_rule_id      VARCHAR(50) NOT NULL,
    reward_rule_version VARCHAR(20) NOT NULL,

    -- Calculation outcome
    eligibility_status  VARCHAR(30) NOT NULL,   -- ELIGIBLE | INELIGIBLE
    reward_type         VARCHAR(30) NOT NULL,   -- POINTS | DOLLAR | GIFT_CARD | etc.
    reward_amount       NUMERIC(10,2) NOT NULL DEFAULT 0,

    -- Calculation reasoning (VERY IMPORTANT)
    calculation_reason  VARCHAR(255) NULL,
    calculation_details JSONB NULL,

    -- Execution context
    execution_id        UUID NOT NULL,
    correlation_id      UUID NOT NULL,
    calculated_at       TIMESTAMP WITH TIME ZONE NOT NULL,

    -- Issuance linkage (filled later)
    issuance_status     VARCHAR(30) NOT NULL DEFAULT 'NOT_ISSUED',
    issuance_id         UUID NULL,

    -- Audit
    created_at          TIMESTAMP WITH TIME ZONE DEFAULT now(),

    -- Idempotency guarantee
    CONSTRAINT uq_reward_calc
        UNIQUE (
            member_activity_id,
            reward_rule_id,
            reward_rule_version
        )
);
